#include <iostream>
#include <math.h>
using namespace std;
int main()
{
	int x;
	int y;
	int C;
	float temp;
	x = 20;
	y = 5
	C = 0;
	temp = 0;
	temp = cos(y);
	temp = pow(temp,3);
	C = 3 * x;
	C = C + temp;
	temp = x * y;
	temp = 2 * temp;
	temp = sin(temp);
	C = C / temp;
	cout << C << endl;
	return 0;
}
